SELECT * FROM DEPARTMENTS D1 
			WHERE 1 = 								
					(SELECT CASE WHEN AVG(SALARY) > 1000 THEN 1 
						ELSE 0 
						END AS STATUS
								FROM 
								EMPLOYEES E1 
								WHERE 
									E1.DEPARTMENT_ID = D1.DEPARTMENT_ID);
									
									
			
		SELECT * FROM DEPARTMENTS 
			WHERE DEPARTMENT_ID IN 
				(SELECT DEPARTMENT_ID FROM (
					(SELECT DEPARTMENT_ID, AVG(SALARY) SAL 
						FROM EMPLOYEES 
							GROUP BY DEPARTMENT_ID)) T 
									WHERE T.SAL > 1000);
